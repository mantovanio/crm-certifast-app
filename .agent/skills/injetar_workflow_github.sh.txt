#!/bin/bash
# Cria a pasta do GitHub Actions se não existir
mkdir -p .github/workflows

# Cria o arquivo de configuração de varredura de segurança
cat << 'EOF' > .github/workflows/security.yml
name: "Varredura de Segurança Autônoma"

on:
  push:
    branches: [ "main", "develop" ]
  pull_request:
    branches: [ "main" ]

jobs:
  analyze:
    name: CodeQL (Análise de Código)
    runs-on: ubuntu-latest
    permissions:
      actions: read
      contents: read
      security-events: write

    steps:
    - name: Checkout repository
      uses: actions/checkout@v4

    - name: Initialize CodeQL
      uses: github/codeql-action/init@v3
      with:
        languages: 'typescript'

    - name: Perform CodeQL Analysis
      uses: github/codeql-action/analyze@v3
EOF

echo "Workflow de segurança do GitHub CodeQL injetado com sucesso!"
